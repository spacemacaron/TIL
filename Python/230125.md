``` py

#튜플
my_tuple1 = ()
print(my_tuple1)
#()

my_tuple2 = (1, -2, 3.14)
print(my_tuple2)
#(1, -2, 3.14)

my_tuple3 = '앨리스', 10, 1.0, 1.2
print(my_tuple3)
#('앨리스', 10, 1.0, 1.2)


#값이 한 개인 튜플을 만드는 코드

print(type(my_int))
#<class 'int'>

my_tuple = (1,)
print(type(my_tuple))
#<class 'tuple'>
```
