``` py


# while 로 반복하는 가위바위보 게임

import random
# 컴퓨터가 무작위로 선택하게 하는 데 필요

rps = ['가위', '바위', '보']
# 리스트에 가위 바위 보를 저장

# 게임을 무한반복 
while True:
    player = input('가위/바위/보/끝:  ')
    computer = random.choice(rps)

    if player == '끝':
        break

    print(player, computer)


    if player == computer:
        print('비겼어요!')
    elif player == '가위':
        
        if computer == '바위':
            print('졌어요!')
        else:
            print('이겼어요!')
    elif player == '바위':
        if computer == '보':
            print('이겼어요!')
    elif player == '보':
        if computer == '가위':
            print('졌어요!')
        else:
            print('이겼어요!')
            



```
